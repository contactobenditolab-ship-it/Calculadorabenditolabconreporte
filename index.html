// Function to build the client budget email content function buildClientBudgetEmail(budgetDetails) { const { clientName, totalAmount, items } = budgetDetails; let emailContent = `Hello ${clientName},\n\nHere are the details of your budget:\n`; items.forEach(item => { emailContent += `- ${item.description}: $${item.amount}\n`; }); emailContent += `Total Amount: $${totalAmount}\n\nThank you!`; return emailContent; } // Function to send the budget email function sendBudgetEmail(budgetDetails) { const emailContent = buildClientBudgetEmail(budgetDetails); // Simulate sending an email console.log('Sending email:', emailContent); } // Set up event listeners for email buttons document.getElementById('btn_email_presupuesto').addEventListener('click', () => { const budgetDetails = { clientName: 'John Doe', totalAmount: 500, items: [ { description: 'Service A', amount: 250 }, { description: 'Service B', amount: 250 } ]}; sendBudgetEmail(budgetDetails); }); // Assuming btn_guardar_notion functionality document.getElementById('btn_guardar_notion').addEventListener('click', () => { console.log('Button to save to Notion clicked'); });
